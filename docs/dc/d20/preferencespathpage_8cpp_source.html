144//---------------------------------------------------------------------------------------------------------------------
145void PreferencesPathPage::editPath()
146{
147 const int row = ui->pathTable->currentRow();
148 QTableWidgetItem *item = ui->pathTable->item(row, 1);
149 SCASSERT(item != nullptr)
150
151 QString path;
152 switch (row)
153 {
154 case 0: // pattern path
155 path = qApp->Seamly2DSettings()->getPatternPath();
156 break;
157 case 1: // templates
158 path = qApp->Seamly2DSettings()->getTemplatePath();
159 break;
160 case 2: // individual measurements
161 path = qApp->Seamly2DSettings()->getIndividualSizePath();
162 break;
163 case 3: // multisize measurements
164 path = qApp->Seamly2DSettings()->getMultisizePath();
165 path = VCommonSettings::prepareMultisizeTables(path);
166 break;
167 case 4: // layout path
168 path = qApp->Seamly2DSettings()->getLayoutPath();
169 break;
170 case 5: // label templates
171 path = qApp->Seamly2DSettings()->getLabelTemplatePath();
172 break;
173 default:
174 break;
175 }
176
177 bool usedNotExistedDir = false;
178 QDir directory(path);
179 if (not directory.exists())
180 {
181 usedNotExistedDir = directory.mkpath(".");
182 }
183
184 QString filename = fileDialog(this, tr("Open Directory"), path, QString(""), nullptr,
185 QFileDialog::ShowDirsOnly |
186 QFileDialog::DontResolveSymlinks |
187 QFileDialog::DontUseNativeDialog,
188 QFileDialog::Directory, QFileDialog::AcceptOpen);
189
190 const QString dir = QFileInfo(filename).filePath();
191
192 if (usedNotExistedDir)
193 {
194 QDir directory(path);
195 directory.rmpath(".");
196 }
197
198 if (dir.isEmpty())
199 {
200 return;
201 }
202
203 item->setText(dir);
204 item->setToolTip(dir);
205}
