170template <typename Dialog>
171/**
172 * @brief ContextMenu show context menu for tool.
173 * @param itemId id of point.
174 * @param event context menu event.
175 * @param showRemove true - tool enable option delete.
176 * @param ref true - do not ignore referens value.
177 */
178void VDrawTool::ContextMenu(QGraphicsSceneContextMenuEvent *event, quint32 itemId,
179 const RemoveOption &showRemove, const Referens &ref)
180{
181 SCASSERT(event != nullptr)
182
183 if (m_suppressContextMenu)
184 {
185 return;
186 }
187
188 GOType itemType = GOType::Unknown;
189 if(itemId != NULL_ID)
190 {
191 try
192 {
193 itemType = data.GetGObject(itemId)->getType();
194 }
195 catch (const VExceptionBadId &error)
196 { // Possible case. Parent was deleted, but the node object is still here.
197 qWarning() << qUtf8Printable(error.ErrorMessage());
198 }
199 }
200
201 quint32 toolId = this->getId();
202 QMap<quint32, Tool> history = doc->getGroupObjHistory();
203 Tool tooltype = history.value(toolId);
204
205 qCDebug(vTool, "Creating tool context menu.");
206 QMenu menu;
207
208 // Actions for copying tool length and angle
209 QAction *actionCopyToolLength = nullptr;
210 QAction *actionCopyLineAngle = nullptr;
211
212 QAction *actionOption = menu.addAction(QIcon::fromTheme("preferences-other"), tr("Properties"));
213
214 // Show object name menu item
215 QAction *actionShowPointName = menu.addAction(QIcon("://icon/16x16/open_eye.png"), tr("Show Point Name"));
216 actionShowPointName->setCheckable(true);
217
218 if (itemType == GOType::Point)
219 {
220 actionShowPointName->setChecked(isPointNameVisible(itemId));
221 }
222 else
223 {
224 actionShowPointName->setVisible(false);
225 }
226
227 // Add Copy menu
228 QMenu *copyMenu = menu.addMenu(QIcon("://icon/32x32/clipboard_icon.png"), tr("Copy"));
229 actionCopyToolLength = copyMenu->addAction(tr("Length"));
230 actionCopyLineAngle = copyMenu->addAction(tr("Angle"));
231
232 // Only add the Angle submneu for the point tools that add a line.
233 if (tooltype != Tool::Line &&
234 tooltype != Tool::EndLine &&
235 tooltype != Tool::Bisector &&
236 tooltype != Tool::Height &&
237 tooltype != Tool::Normal &&
238 tooltype != Tool::LineIntersectAxis &&
239 tooltype != Tool::AlongLine &&
240 tooltype != Tool::CurveIntersectAxis)
241 {
242 actionCopyLineAngle->setVisible(false);
243 }
244
245 QAction *actionDelete = menu.addAction(QIcon::fromTheme("edit-delete"), tr("Delete"));
246 if (showRemove == RemoveOption::Enable)
247 {
248 if (ref == Referens::Follow)
249 {
250 if (_referens > 1)
251 {
252 qCDebug(vTool, "Delete disabled. Tool has children.");
253 actionDelete->setEnabled(false);
254 }
255 else
256 {
257 qCDebug(vTool, "Delete enabled. Tool has no children.");
258 actionDelete->setEnabled(true);
259 }
260 }
261 else
262 {
263 qCDebug(vTool, "Delete enabled. Ignore referens value.");
264 actionDelete->setEnabled(true);
265 }
266 }
267 else
268 {
269 qCDebug(vTool, "Delete disabled.");
270 actionDelete->setEnabled(false);
271 }
272
273 // Add Group Item menu item
274 QMap<quint32,QString> groupsNotContainingItem = doc->getGroupsContainingItem(this->getId(), itemId, false);
275 QActionGroup* actionAddGroupMenu= new QActionGroup(this);
276
277 if (!groupsNotContainingItem.empty())
278 {
279 QMenu *menuAddGroupItem = menu.addMenu(QIcon("://icon/32x32/add.png"), tr("Add Group Object"));
280 QStringList list = QStringList(groupsNotContainingItem.values());
281 list.sort(Qt::CaseInsensitive);
282
283 for(int i=0; i<list.count(); ++i)
284 {
285 QAction *actionAddGroupItem = menuAddGroupItem->addAction(list[i]);
286 actionAddGroupMenu->addAction(actionAddGroupItem);
287 const quint32 groupId = groupsNotContainingItem.key(list[i]);
288 actionAddGroupItem->setData(groupId);
289 groupsNotContainingItem.remove(groupId); // delete any duplicate groups
290 }
291 }
292
293 // Remove Group Item menu item
294 QMap<quint32,QString> groupsContainingItem = doc->getGroupsContainingItem(this->getId(), itemId, true);
295 QActionGroup* actionDeleteGroupMenu = new QActionGroup(this);
296
297 if (!groupsContainingItem.empty())
298 {
299 QMenu *menuRemoveGroupItem = menu.addMenu(QIcon("://icon/32x32/remove.png"), tr("Remove Group Object"));
300
301 QStringList list = QStringList(groupsContainingItem.values());
302 list.sort(Qt::CaseInsensitive);
303
304 for(int i=0; i<list.count(); ++i)
305 {
306 QAction *actionRemoveGroupItem = menuRemoveGroupItem->addAction(list[i]);
307 actionDeleteGroupMenu->addAction(actionRemoveGroupItem);
308 const quint32 groupId = groupsContainingItem.key(list[i]);
309 actionRemoveGroupItem->setData(groupId);
310 groupsContainingItem.remove(groupId);
311 }
312 }
313
314 QAction *selectedAction = menu.exec(event->screenPos());
315 if(selectedAction == nullptr)
316 {
317 return;
318 }
319 if (selectedAction == actionOption)
320 {
321 qCDebug(vTool, "Show options.");
322 qApp->getSceneView()->itemClicked(nullptr);
323 m_dialog = QSharedPointer<Dialog>(new Dialog(getData(), m_id, qApp->getMainWindow()));
324 m_dialog->setModal(true);
325
326 connect(m_dialog.data(), &DialogTool::DialogClosed, this, &VDrawTool::FullUpdateFromGuiOk);
327 connect(m_dialog.data(), &DialogTool::DialogApplied, this, &VDrawTool::FullUpdateFromGuiApply);
328
329 this->setDialog();
330
331 m_dialog->show();
332 }
333 else if (selectedAction == actionDelete)
334 {
335 qCDebug(vTool, "Deleting Object.");
336 deleteTool(); // do not catch exception here
337 return; //Leave this method immediately after call!!!
338 }
339 else if (selectedAction == actionShowPointName)
340 {
341 updatePointNameVisibility(itemId, selectedAction->isChecked());
342 }
343
344 else if (selectedAction == actionCopyToolLength)
345 {
346 QString text = QString("");
347
348 switch (tooltype)
349 {
350 case Tool::Line:
351 {
352 const QDomElement domElement = doc->elementById(toolId);
353 if (domElement.isElement())
354 {
355 text = tr("Line_") +
356 data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, AttrFirstPoint, "0"))->name() +
357 "_"+
358 data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, AttrSecondPoint, "0"))->name();
359 break;
360 }
361 }
362 case Tool::Arc:
363 case Tool::ArcWithLength:
364 {
365 const QSharedPointer<VArc> arc = data.GeometricObject<VArc>(toolId);
366 SCASSERT(!arc.isNull())
367 text = arc->NameForHistory(tr("Arc_"));
368 break;
369 }
370 case Tool::ShoulderPoint:
371 case Tool::Normal:
372 case Tool::Bisector:
373 case Tool::LineIntersect:
374 case Tool::BasePoint:
375 case Tool::EndLine:
376 case Tool::PointOfContact:
377 case Tool::Height:
378 case Tool::Triangle:
379 case Tool::PointOfIntersection:
380 case Tool::CutArc:
381 case Tool::CutSpline:
382 case Tool::CutSplinePath:
383 case Tool::LineIntersectAxis:
384 case Tool::CurveIntersectAxis:
385 case Tool::PointOfIntersectionArcs:
386 case Tool::PointOfIntersectionCircles:
387 case Tool::PointOfIntersectionCurves:
388 case Tool::PointFromCircleAndTangent:
389 case Tool::PointFromArcAndTangent:
390 {
391 const QSharedPointer<VGObject> obj = data.GetGObject(toolId);
392 SCASSERT(!obj.isNull())
393 text = obj->name();
394 break;
395 }
396
397 case Tool::EllipticalArc:
398 {
399 const QSharedPointer<VEllipticalArc> elArc = data.GeometricObject<VEllipticalArc>(toolId);
400 SCASSERT(!elArc.isNull())
401 text = elArc->NameForHistory(tr("ElArc_"));
402 break;
403 }
404
405 case Tool::Spline:
406 {
407 const QSharedPointer<VSpline> spl = data.GeometricObject<VSpline>(toolId);
408 SCASSERT(!spl.isNull())
409 text = spl->NameForHistory(tr("Spl_"));
410 break;
411 }
412
413 case Tool::SplinePath:
414 {
415 const QSharedPointer<VSplinePath> splPath = data.GeometricObject<VSplinePath>(toolId);
416 SCASSERT(!splPath.isNull())
417 text = splPath->NameForHistory(tr("SplPath_"));
418 break;
419 }
420
421 case Tool::CubicBezier:
422 {
423 const QSharedPointer<VCubicBezier> spl = data.GeometricObject<VCubicBezier>(toolId);
424 SCASSERT(!spl.isNull())
425 text = spl->NameForHistory(tr("Spl_"));
426 break;
427 }
428
429 case Tool::CubicBezierPath:
430 {
431 const QSharedPointer<VCubicBezierPath> splPath = data.GeometricObject<VCubicBezierPath>(toolId);
432 SCASSERT(!splPath.isNull())
433 text = splPath->NameForHistory(tr("SplPath_"));
434 break;
435 }
436
437 case Tool::Rotation:
438 case Tool::Move:
439 case Tool::MirrorByLine:
440 case Tool::MirrorByAxis:
441 {
442 const QSharedPointer<VGObject> obj = data.GetGObject(itemId);
443 SCASSERT(!obj.isNull())
444 text = obj->name();
445 break;
446 }
447 }
448
449 QClipboard *clipboard = QApplication::clipboard();
450 clipboard->setText(text);
451 }
452 else if (selectedAction->actionGroup() == actionAddGroupMenu)
453 {
454 quint32 groupId = selectedAction->data().toUInt();
455 QDomElement item = doc->addGroupItem(this->getId(), itemId, groupId);
456
457 VMainGraphicsScene *scene = qobject_cast<VMainGraphicsScene *>(qApp->getCurrentScene());
458 SCASSERT(scene != nullptr)
459 scene->clearSelection();
460
461 VAbstractMainWindow *window = qobject_cast<VAbstractMainWindow *>(qApp->getMainWindow());
462 SCASSERT(window != nullptr)
463 {
464 AddGroupItem *command = new AddGroupItem(item, doc, groupId);
465 connect(command, &AddGroupItem::updateGroups, window, &VAbstractMainWindow::updateGroups);
466 qApp->getUndoStack()->push(command);
467 }
468 }
469 else if (selectedAction == actionCopyLineAngle)
470 {
471 QString angleName = QString("");
472
473 switch (tooltype)
474 {
475 case Tool::Line:
476 {
477 const QDomElement domElement = doc->elementById(toolId);
478 if (domElement.isElement())
479 {
480 angleName = tr("AngleLine_") +
481 data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, AttrFirstPoint, "0"))->name() +
482 "_"+
483 data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, AttrSecondPoint, "0"))->name();
484 break;
485 }
486 }
487 case Tool::AlongLine:
488 case Tool::Normal:
489 {
490 const QDomElement domElement = doc->elementById(toolId);
491 if (domElement.isElement())
492 {
493 angleName = tr("AngleLine_") +
494 data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, AttrFirstPoint, "0"))->name() +
495 "_" + data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, "id", "0"))->name();
496 break;
497 }
498 }
499 case Tool::Bisector:
500 {
501 const QDomElement domElement = doc->elementById(toolId);
502 if (domElement.isElement())
503 {
504 angleName = tr("AngleLine_") +
505 data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, AttrSecondPoint, "0"))->name() +
506 "_" + data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, "id", "0"))->name();
507 break;
508 }
509 }
510 case Tool::EndLine:
511 case Tool::Height:
512 case Tool::LineIntersectAxis:
513 case Tool::CurveIntersectAxis:
514 {
515 const QDomElement domElement = doc->elementById(toolId);
516 if (domElement.isElement())
517 {
518 angleName = tr("AngleLine_") +
519 data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, AttrBasePoint, "0"))->name() +
520 "_" + data.GeometricObject<VPointF>(doc->GetParametrUInt(domElement, "id", "0"))->name();
521 break;
522 }
523 }
524 }
525 QClipboard *clipboard = QApplication::clipboard();
526 clipboard->setText(angleName);
527 }
528
529 else if (selectedAction->actionGroup() == actionDeleteGroupMenu)
530 {
531 quint32 groupId = selectedAction->data().toUInt();
532 QDomElement item = doc->removeGroupItem(this->getId(), itemId, groupId);
533
534 VMainGraphicsScene *scene = qobject_cast<VMainGraphicsScene *>(qApp->getCurrentScene());
535 SCASSERT(scene != nullptr)
536 scene->clearSelection();
537
538 VAbstractMainWindow *window = qobject_cast<VAbstractMainWindow *>(qApp->getMainWindow());
539 SCASSERT(window != nullptr)
540 {
541 RemoveGroupItem *command = new RemoveGroupItem(item, doc, groupId);
542 connect(command, &RemoveGroupItem::updateGroups, window, &VAbstractMainWindow::updateGroups);
543 qApp->getUndoStack()->push(command);
544 }
545 }
