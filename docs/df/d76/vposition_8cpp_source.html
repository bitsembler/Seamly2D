103/// @brief run run the thread to position piece.
104///
105/// This method runs the thread that positions the piece near to the sheet edge.
106//---------------------------------------------------------------------------------------------------------------------
107void VPosition::run()
108{
109 if (m_stop->load())
110 {
111 return;
112 }
113
114 // We should use copy of the piece.
115 VLayoutPiece workpiece = m_piece;
116
117 int pieceEdgeNum = m_pieceEdgeNum; // For flipped piece edge will be different
118 if (edgesIntersect(workpiece, pieceEdgeNum, m_sheetEdgeNum))
119 {
120 saveCandidate(m_bestResult, workpiece, pieceEdgeNum, m_sheetEdgeNum, BestFrom::Combine);
121 }
122
123 if (m_rotate)
124 {
125 rotate(m_rotationIncrement);
126 }
127 else
128 {
129 if (m_sheet.GetContour().isEmpty())
130 {
131 rotate(m_rotationIncrement);
132 }
133 }
134}

170/// @brief edgesIntersect check for intersection between piece and sheet
171///
172/// This method checks if the pattern piece edge intersects with the sheet edge.
173///
174/// @param piece pattern piece being added to sheet.
175/// @param pieceEdgeNum edge number of piece to be combined with the sheet edge.
176/// @param sheetEdgeNum edge of sheet piece is being comibed with.
177/// @returns bool.
178/// - true if there no intersection.
179/// - false if it has an intersection.
180//---------------------------------------------------------------------------------------------------------------------
181bool VPosition::edgesIntersect(VLayoutPiece &piece, int &pieceEdgeNum, int sheetEdgeNum)
182{
183 const QLineF sheetEdge = m_sheet.GlobalEdge(sheetEdgeNum);
184 bool isFlipped = false;
185 bool hasNoIntersectiom = false;
186
187 combineEdges(piece, pieceEdgeNum, sheetEdge);
188
189 IntersectionType type = IntersectionType::Intersection;
190 if (sheetContains(piece.pieceBoundingRect()))
191 {
192 if (!m_sheet.GetContour().isEmpty())
193 {
194 type = intersectionType(piece);
195 }
196 else
197 {
198 type = IntersectionType::NoIntersection;
199 }
200 }
201
202 switch (type)
203 {
204 case IntersectionType::Intersection:
205 piece.Mirror(sheetEdge);
206 isFlipped = true;
207 break;
208 case IntersectionType::NoIntersection:
209 hasNoIntersectiom = true;
210 break;
211 default:
212 break;
213 }
214
215 if (isFlipped && !piece.IsForbidFlipping())
216 {
217 if (m_sheet.GetContour().isEmpty())
218 {
219 pieceEdgeNum = piece.pieceEdgeByPoint(sheetEdge.p2());
220 }
221 else
222 {
223 pieceEdgeNum = piece.LayoutEdgeByPoint(sheetEdge.p2());
224 }
225
226 if (pieceEdgeNum <= 0)
227 {
228 return false;
229 }
230
231 IntersectionType type = IntersectionType::Intersection;
232 if (sheetContains(piece.pieceBoundingRect()))
233 {
234 type = intersectionType(piece);
235 }
236
237 switch (type)
238 {
239 case IntersectionType::Intersection:
240 hasNoIntersectiom = false;
241 break;
242 case IntersectionType::NoIntersection:
243 hasNoIntersectiom = true;
244 break;
245 default:
246 break;
247 }
248 }
249 return hasNoIntersectiom;
250}

291//---------------------------------------------------------------------------------------------------------------------
292/// @brief intersectionType get the intersection type between sheet and piece.
293///
294/// This method checks if the bounding box of a piece intersects with sheet bounding box.
295///
296/// @param piece layput piece.
297/// @returns IntersectionType intersection type
298/// - NoIntersection = 0,
299/// - Intersection = 1,
300//---------------------------------------------------------------------------------------------------------------------
301VPosition::IntersectionType VPosition::intersectionType(const VLayoutPiece &piece) const
302{
303 const QRectF globalRect = m_sheet.BoundingRect();
304 if (!globalRect.intersects(piece.LayoutBoundingRect()) && !globalRect.contains(piece.pieceBoundingRect()))
305 {
306 // This we can determine efficiently.
307 return IntersectionType::NoIntersection;
308 }
309
310 const QPainterPath globalPath = m_sheet.ContourPath();
311 if (!globalPath.intersects(piece.LayoutAllowancePath()) && !globalPath.contains(piece.createMainPath()))
312 {
313 return IntersectionType::NoIntersection;
314 }
315 else
316 {
317 return IntersectionType::Intersection;
318 }
319}

337//---------------------------------------------------------------------------------------------------------------------
338/// @brief combineEdges rotate and move piece to contour edge.
339///
340/// This method rotates and moves piece near to the sheet edge.
341///
342/// @param piece layout piece.
343/// @param pieceEdgeNum piece edge number.
344/// @param sheetEdge line that represents the sheet edge.
345/// @returns void.
346//---------------------------------------------------------------------------------------------------------------------
347void VPosition::combineEdges(VLayoutPiece &piece, const int &pieceEdgeNum, const QLineF &sheetEdge)
348{
349 QLineF pieceEdge;
350 if (m_sheet.GetContour().isEmpty())
351 {
352 pieceEdge = piece.pieceEdge(pieceEdgeNum);
353 }
354 else
355 {
356 pieceEdge = piece.LayoutEdge(pieceEdgeNum);
357 }
358
359 // Find distance between sheet and piece edges.
360 const qreal dx = sheetEdge.x2() - pieceEdge.x2();
361 const qreal dy = sheetEdge.y2() - pieceEdge.y2();
362
363 pieceEdge.translate(dx, dy); // Use values for translate piece edge.
364
365 m_angleBetween = sheetEdge.angleTo(pieceEdge); // Seek angle between two edges.
366
367 // Move piece to position near to sheet edge.
368 piece.Translate(dx, dy);
369 if (!qFuzzyIsNull(m_angleBetween) || !qFuzzyCompare(m_angleBetween, 360))
370 {
371 piece.Rotate(pieceEdge.p2(), -m_angleBetween);
372 }
373}

387{

389 if (m_sheet.GetContour().isEmpty())
390 {
391 pieceEdge = piece.pieceEdge(pieceEdgeNum);

397
398 // Find distance between sheet and piece edges.
399 const qreal dx = sheetEdge.x2() - pieceEdge.x2();
400 const qreal dy = sheetEdge.y2() - pieceEdge.y2();
401
402 // Move and rotate piece to position near to sheet edge.
403 piece.Translate(dx, dy);
404 piece.Rotate(sheetEdge.p2(), angle);

407//---------------------------------------------------------------------------------------------------------------------
408/// @brief rotate rotate the pattern piece.
409///
410/// This method rotates the pattern piece by the number of degrees in incrmentuntil it
411/// intersects with the sheet. The best square result is saved.
412///
413/// @param increment pattern piece rotation angle.
414/// @return void.
415//---------------------------------------------------------------------------------------------------------------------
416void VPosition::rotate(int increment)
417{
418 int startAngle = 0;
419 if (VFuzzyComparePossibleNulls(m_angleBetween, 360))
420 {
421 startAngle = increment;
422 }
423 for (int angle = startAngle; angle < 360; angle = angle + increment)
424 {
425 if (m_stop->load())
426 {
427 return;
428 }
429
430 // We should use copy of the piece.
431 VLayoutPiece workpiece = m_piece;
432
433 if (rotatedEdgeIntersects(workpiece, m_pieceEdgeNum, m_sheetEdgeNum, angle))
434 {
435 saveCandidate(m_bestResult, workpiece, m_pieceEdgeNum, m_sheetEdgeNum, BestFrom::Rotation);
436 }
437 }
