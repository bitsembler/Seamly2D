118/// @brief VGrainlineItem::paint paints the item content
119/// @param painter pointer to the painter object
120/// @param option not used
121/// @param widget not used
122//---------------------------------------------------------------------------------------------------------------------
123void VGrainlineItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
124{
125 Q_UNUSED(option)
126 Q_UNUSED(widget)
127
128 painter->save();
129 QColor color = QColor(qApp->Settings()->getDefaultGrainlineColor());
130 m_penWidth = ToPixel(qApp->Settings()->getDefaultGrainlineLineweight(), Unit::Mm) * 3;
131 painter->setPen(QPen(color, m_penWidth, Qt::SolidLine, Qt::FlatCap, Qt::MiterJoin));
132
133 painter->setRenderHints(QPainter::Antialiasing);
134 // line
135 const QLineF line = mainLine();
136 painter->drawLine(line.p1(), line.p2());
137
138 painter->setBrush(color);
139
140 if (m_arrowType != ArrowType::Bottom)
141 {
142 // first arrow
143 painter->drawPolygon(firstArrow());
144 }
145 if (m_arrowType != ArrowType::Top)
146 {
147 // second arrow
148 painter->drawPolygon(secondArrow());
149 }
150
151 if (m_mode != Mode::Normal)
152 {
153 painter->setPen(QPen(Qt::black, 2, Qt::DashLine));
154 painter->setBrush(Qt::NoBrush);
155 // bounding polygon
156 painter->drawPolygon(m_boundingPoly);
157
158 if (m_mode != Mode::Rotate)
159 {
160 painter->setPen(QPen(Qt::black, 3));
161 painter->setBrush(Qt::black);
162 updateResizeHandle();
163 painter->drawPolygon(m_resizeHandle);
164 }
165
166 painter->setBrush(Qt::NoBrush);
167 if (m_mode == Mode::Resize)
168 {
169 painter->setPen(Qt::black);
170 painter->drawLine(m_boundingPoly.at(0), m_boundingPoly.at(2));
171 painter->drawLine(m_boundingPoly.at(1), m_boundingPoly.at(3));
172 }
173
174 if (m_mode == Mode::Rotate)
175 {
176 QPointF ptC = (m_boundingPoly.at(0) + m_boundingPoly.at(2))/2;
177 qreal dRad = ROTATE_CIRC_R;
178 painter->setBrush(Qt::black);
179 painter->drawEllipse(ptC, dRad, dRad);
180
181 painter->setBrush(Qt::NoBrush);
182 painter->save();
183 painter->translate(ptC);
184 painter->rotate(qRadiansToDegrees(-m_rotation));
185 int iX = int(qRound(m_length/2 - 0.5*dRad));
186 int iY = int(qRound(RECT_WIDTH - 0.5*dRad));
187 int iR = int(qRound(dRad*3));
188 painter->drawArc(iX - iR, iY - iR, iR, iR, 0*16, -90*16);
189 painter->drawArc(-iX, iY - iR, iR, iR, 270*16, -90*16);
190 painter->drawArc(-iX, -iY, iR, iR, 180*16, -90*16);
191 painter->drawArc(iX - iR, -iY, iR, iR, 90*16, -90*16);
192 painter->restore();
193 }
194 }
195 painter->restore();
196}

199/// @brief VGrainlineItem::updateGeometry updates the item with grainline parameters.
200/// @param position position of one grainline's end.
201/// @param rotation rotation of the grainline in [degrees].
202/// @param length length of the grainline in user's units.
203/// @param type type of arrowhead.
204//---------------------------------------------------------------------------------------------------------------------
205void VGrainlineItem::updateGeometry(const QPointF& position, qreal rotation, qreal length, ArrowType type)
206{
207 m_rotation = qDegreesToRadians(rotation);
208 m_length = length;
209
210 qreal dX;
211 qreal dY;
212 QPointF point = position;
213 if (isContained(point, m_rotation, dX, dY) == false)
214 {
215 point.setX(point.x() + dX);
216 point.setY(point.y() + dY);
217 }
218 setPos(point);
219 m_arrowType = type;
220
221 updateRectangle();
222 updateItem();
223}

225//---------------------------------------------------------------------------------------------------------------------
226/// @brief VGrainlineItem::isContained checks, if both ends of the grainline, starting at point, are contained in
227/// parent widget.
228/// @param point starting point of the grainline.
229/// @param rotation rotation of the grainline in [rad]
230/// @param dX horizontal translation needed to put the arrow inside parent item
231/// @param dY vertical translation needed to put the arrow inside parent item
232/// @return true, if both ends of the grainline, starting at point, are contained in the parent widget and
233/// false otherwise.
234//---------------------------------------------------------------------------------------------------------------------
235bool VGrainlineItem::isContained(const QPointF& point, qreal rotation, qreal &dX, qreal &dY) const
236{
237 dX = 0;
238 dY = 0;
239 QPointF apoint[2];
240 apoint[0] = point;
241 apoint[1].setX(point.x() + m_length * cos(rotation));
242 apoint[1].setY(point.y() - m_length * sin(rotation));
243 // single point differences
244 qreal pointX;
245 qreal pointY;
246 bool bInside = true;
247
248 QRectF rectParent = parentItem()->boundingRect();
249 for (int i = 0; i < 2; ++i)
250 {
251 pointX = 0;
252 pointY = 0;
253 if (rectParent.contains(apoint[i]) == false)
254 {
255 if (apoint[i].x() < rectParent.left())
256 {
257 pointX = rectParent.left() - apoint[i].x();
258 }
259 else if (apoint[i].x() > rectParent.right())
260 {
261 pointX = rectParent.right() - apoint[i].x();
262 }
263 if (apoint[i].y() < rectParent.top())
264 {
265 pointY = rectParent.top() - apoint[i].y();
266 }
267 else if (apoint[i].y() > rectParent.bottom())
268 {
269 pointY = rectParent.bottom() - apoint[i].y();
270 }
271
272 if (fabs(pointX) > fabs(dX))
273 {
274 dX = pointX;
275 }
276 if (fabs(pointY) > fabs(dY))
277 {
278 dY = pointY;
279 }
280
281 bInside = false;
282 }
283 }
284 return bInside;
285}

288/// @brief VGrainlineItem::mousePressEvent handles left button mouse press events
289/// @param event pointer to QGraphicsSceneMouseEvent object
290//---------------------------------------------------------------------------------------------------------------------
291void VGrainlineItem::mousePressEvent(QGraphicsSceneMouseEvent* event)
292{
293 if (event->button() == Qt::LeftButton && event->type() != QEvent::GraphicsSceneMouseDoubleClick
294 && (flags() & QGraphicsItem::ItemIsMovable))
295 {
296 if (m_moveType == NotMovable)
297 {
298 event->ignore();
299 return;
300 }
301
302 m_startPos = pos();
303 m_movePos = event->scenePos();
304 m_startLength = m_length;
305 m_rotationStart = m_rotation;
306 m_angle = GetAngle(mapToParent(event->pos()));
307 m_rotationCenter = m_centerPoint;
308
309 if ((m_moveType & AllModifications ) == AllModifications)
310 {
311 allUserModifications(event->pos());
312 setZValue(ACTIVE_Z);
313 updateItem();
314 }
315 else if (m_moveType & IsRotatable)
316 {
317 if (m_moveType & IsResizable)
318 {
319 allUserModifications(event->pos());
320 }
321 else if (m_moveType & IsMovable)
322 {
323 userRotateAndMove();
324 }
325 else
326 {
327 m_mode = Mode::Rotate;
328 SetItemOverrideCursor(this, cursorArrowCloseHand, 1, 1);
329 }
330 setZValue(ACTIVE_Z);
331 updateItem();
332 }
333 else if (m_moveType & IsResizable)
334 {
335 if (m_moveType & IsRotatable)
336 {
337 allUserModifications(event->pos());
338 }
339 else if (m_moveType & IsMovable)
340 {
341 userMoveAndResize(event->pos());
342 }
343 setZValue(ACTIVE_Z);
344 updateItem();
345 }
346 else if (m_moveType & IsMovable)
347 {
348 if (m_moveType & IsRotatable)
349 {
350 userRotateAndMove();
351 }
352 else if (m_moveType & IsResizable)
353 {
354 userMoveAndResize(event->pos());
355 }
356 else
357 {
358 m_mode = Mode::Move;
359 SetItemOverrideCursor(this, cursorArrowCloseHand, 1, 1);
360 }
361
362 setZValue(ACTIVE_Z);
363 updateItem();
364 }
365 else
366 {
367 event->ignore();
368 return;
369 }
370 }
371}

373//---------------------------------------------------------------------------------------------------------------------
374/// @brief VGrainlineItem::mouseMoveEvent handles mouse move events, making sure that the item is moved properly
375/// @param event pointer to QGraphicsSceneMouseEvent object
376//---------------------------------------------------------------------------------------------------------------------
377void VGrainlineItem::mouseMoveEvent(QGraphicsSceneMouseEvent* event)
378{
379 QPointF delta = event->scenePos() - m_movePos;
380 qreal dX;
381 qreal dY;
382 if (m_mode == Mode::Move && m_moveType & IsMovable)
383 {
384 QPointF point = m_startPos + delta;
385 if (isContained(point, m_rotation, dX, dY) == false)
386 {
387 point.setX(point.x() + dX);
388 point.setY(point.y() + dY);
389 }
390 setPos(point);
391 updateItem();
392 }
393 else if (m_mode == Mode::Resize && m_moveType & IsResizable)
394 {
395 qreal deltaLength = qSqrt(delta.x()*delta.x() + delta.y()*delta.y());
396 qreal deltaAngle = qAtan2(-delta.y(), delta.x());
397 deltaLength = -deltaLength*qCos(deltaAngle - m_rotation);
398 qreal deltaPrevLength = m_length;
399 // try with new length
400 if (!(m_moveType & IsMovable))
401 {
402 deltaLength *= 2;
403 }
404 // limit length of grainline to twice the length of the arrowheads.
405 qreal minLength = qApp->Settings()->getDefaultArrowLength() * 2.0;
406 if (m_startLength + deltaLength < minLength)
407 {
408 return;
409 }
410 m_length = m_startLength + deltaLength;
411
412 QPointF pos;
413
414 if (m_moveType & IsMovable)
415 {
416 QLineF grainline(this->pos().x(), this->pos().y(),
417 this->pos().x() + deltaPrevLength, this->pos().y());
418 grainline.setAngle(qRadiansToDegrees(m_rotation));
419 grainline = QLineF(grainline.p2(), grainline.p1());
420 grainline.setLength(m_length);
421 pos = grainline.p2();
422 }
423 else
424 {
425 QLineF grainline(m_centerPoint.x(), m_centerPoint.y(),
426 m_centerPoint.x() + m_length / 2.0, m_centerPoint.y());
427
428 grainline.setAngle(qRadiansToDegrees(m_rotation));
429 grainline = QLineF(grainline.p2(), grainline.p1());
430 grainline.setLength(m_length);
431
432 pos = grainline.p2();
433 }
434
435 qreal dX;
436 qreal dY;
437 if (isContained(pos, m_rotation, dX, dY) == false)
438 {
439 m_length = deltaPrevLength;
440 }
441 else
442 {
443 setPos(pos);
444 }
445
446 updateRectangle();
447 updateItem();
448 }
449 else if (m_mode == Mode::Rotate && m_moveType & IsRotatable)
450 {
451 // prevent strange angle changes due to singularities
452 qreal deltaLength = qSqrt(delta.x()*delta.x() + delta.y()*delta.y());
453 if (deltaLength < 2)
454 {
455 return;
456 }
457
458 if (fabs(m_angle) < 0.01)
459 {
460 m_angle = GetAngle(mapToParent(event->pos()));
461 return;
462 }
463
464 qreal angle = GetAngle(mapToParent(event->pos())) - m_angle;
465 QPointF newPos = rotate(m_startPos, m_rotationCenter, angle);
466 if (isContained(newPos, m_rotationStart + angle, dX, dY) == true)
467 {
468 setPos(newPos);
469 m_rotation = m_rotationStart + angle;
470 updateRectangle();
471 updateItem();
472 }
473 }
474}

476//---------------------------------------------------------------------------------------------------------------------
477/// @brief VGrainlineItem::mouseReleaseEvent handles mouse release events and emits the proper signal if the item was
478/// moved
479/// @param event pointer to QGraphicsSceneMouseEvent object
480//---------------------------------------------------------------------------------------------------------------------
481void VGrainlineItem::mouseReleaseEvent(QGraphicsSceneMouseEvent* event)
482{
483 if (event->button() == Qt::LeftButton)
484 {
485 if ((m_mode == Mode::Move || m_mode == Mode::Rotate || m_mode == Mode::Resize) && (flags() & QGraphicsItem::ItemIsMovable))
486 {
487 SetItemOverrideCursor(this, cursorArrowOpenHand, 1, 1);
488 }
489
490 QPointF delta = event->scenePos() - m_movePos;
491 qreal deltaLength = qSqrt(delta.x()*delta.x() + delta.y()*delta.y());
492 bool bShort = (deltaLength < 2);
493
494 if (m_mode == Mode::Move || m_mode == Mode::Resize)
495 {
496 if (bShort == true)
497 {
498 if (m_isReleased == true && m_moveType & IsRotatable)
499 {
500 m_mode = Mode::Rotate;
501 updateItem();
502 }
503 }
504 else
505 {
506 if (m_mode == Mode::Move && m_moveType & IsMovable)
507 {
508 emit itemMoved(pos());
509 }
510 else if (m_moveType & IsResizable)
511 {
512 emit itemResized(m_length);
513 }
514 updateItem();
515 }
516 }
517 else
518 {
519 if (bShort == true)
520 {
521 m_mode = Mode::Move;
522 }
523 else if (m_moveType & IsRotatable)
524 {
525 emit itemRotated(m_rotation, m_startPoint);
526 }
527 updateItem();
528 }
529 m_isReleased = true;
530 }
531}

555//---------------------------------------------------------------------------------------------------------------------
556/// @brief VGrainlineItem::updateRectangle updates the polygon for the box around active item
557/// and the bounding rectangle
558//---------------------------------------------------------------------------------------------------------------------
559void VGrainlineItem::updateRectangle()
560{
561 QPointF point1(0, 0);
562 QPointF point2(point1.x() + m_length * cos(m_rotation), point1.y() - m_length * sin(m_rotation));
563
564 m_startPoint = mapToParent(point1);
565 m_finishPoint = mapToParent(point2);
566 m_centerPoint = (m_startPoint + m_finishPoint)/2;
567
568 m_boundingPoly.clear();
569 m_boundingPoly << QPointF(point1.x() + RECT_WIDTH*cos(m_rotation + M_PI/2),
570 point1.y() - RECT_WIDTH*sin(m_rotation + M_PI/2));
571 m_boundingPoly << QPointF(point1.x() + RECT_WIDTH*cos(m_rotation - M_PI/2),
572 point1.y() - RECT_WIDTH*sin(m_rotation - M_PI/2));
573 m_boundingPoly << QPointF(point2.x() + RECT_WIDTH*cos(m_rotation - M_PI/2),
574 point2.y() - RECT_WIDTH*sin(m_rotation - M_PI/2));
575 m_boundingPoly << QPointF(point2.x() + RECT_WIDTH*cos(m_rotation + M_PI/2),
576 point2.y() - RECT_WIDTH*sin(m_rotation + M_PI/2));
577 m_boundingRect = m_boundingPoly.boundingRect().adjusted(-2, -2, 2, 2);
578 setTransformOriginPoint(m_boundingRect.center());
579
580 updateResizeHandle();
581 prepareGeometryChange();
582}

590//---------------------------------------------------------------------------------------------------------------------
591/// @brief VGrainlineItem::rotate rotates point around center by angle angle [rad]
592/// and returns the resulting point
593/// @param point point to rotate
594/// @param center center of rotation
595/// @param angle angle of rotation
596/// @return point, which is a result of rotating point around center by angle angle
597//---------------------------------------------------------------------------------------------------------------------
598QPointF VGrainlineItem::rotate(const QPointF& point, const QPointF& center, qreal angle) const
599{
600 QPointF ptRel = point - center;
601 QPointF ptFinal;
602 ptFinal.setX(ptRel.x()*qCos(angle) + ptRel.y()*qSin(angle));
603 ptFinal.setY(-ptRel.x()*qSin(angle) + ptRel.y()*qCos(angle));
604 ptFinal += center;
605 return ptFinal;
606}

608//---------------------------------------------------------------------------------------------------------------------
609/// @brief VGrainlineItem::getInsideCorner calculates a point inside the bounding polygon,
610/// distance away of i-th point in each direction
611/// @param i index of corner
612/// @param distance distance
613/// @return resulting point
614//---------------------------------------------------------------------------------------------------------------------
615QPointF VGrainlineItem::getInsideCorner(int i, qreal distance) const
616{
617 QPointF point1 = m_boundingPoly.at((i + 1) % m_boundingPoly.count()) - m_boundingPoly.at(i);
618
619 QPointF point2 = m_boundingPoly.at((i + m_boundingPoly.count() - 1) %
620 m_boundingPoly.count()) - m_boundingPoly.at(i);
621
622 point1 = distance * point1/qSqrt(point1.x()*point1.x() + point1.y()*point1.y());
623 point2 = distance * point2/qSqrt(point2.x()*point2.x() + point2.y()*point2.y());
624
625 return m_boundingPoly.at(i) + point1 + point2;
626}

636//---------------------------------------------------------------------------------------------------------------------
637QPolygonF VGrainlineItem::firstArrow() const
638{
639 qreal arrowLength = qApp->Settings()->getDefaultArrowLength();
640 QPointF point2 = mainLine().p2();
641 QPolygonF polygon;
642 polygon << point2;
643 polygon << QPointF(point2.x() + arrowLength * cos(M_PI + m_rotation + ARROW_ANGLE),
644 point2.y() - arrowLength * sin(M_PI + m_rotation + ARROW_ANGLE));
645 polygon << QPointF(point2.x() + arrowLength * cos(M_PI + m_rotation - ARROW_ANGLE),
646 point2.y() - arrowLength * sin(M_PI + m_rotation - ARROW_ANGLE));
647 return polygon;
648}

664//---------------------------------------------------------------------------------------------------------------------
665QPainterPath VGrainlineItem::mainShape() const
666{
667 QPainterPath path;
668 const QLineF line = mainLine();
669
670 QPainterPath linePath;
671 linePath.moveTo(line.p1());
672 linePath.lineTo(line.p2());
673 linePath.closeSubpath();
674
675 QPainterPathStroker stroker;
676 stroker.setWidth(m_penWidth);
677 path.addPath((stroker.createStroke(linePath) + linePath).simplified());
678 path.closeSubpath();
679
680 if (m_arrowType != ArrowType::Bottom)
681 {
682 // first arrow
683 QPainterPath polyPath;
684 polyPath.addPolygon(firstArrow());
685 path.addPath((stroker.createStroke(polyPath) + polyPath).simplified());
686 path.closeSubpath();
687 }
688
689 if (m_arrowType != ArrowType::Top)
690 {
691 // second arrow
692 QPainterPath polyPath;
693 polyPath.addPolygon(secondArrow());
694 path.addPath((stroker.createStroke(polyPath) + polyPath).simplified());
695 path.closeSubpath();
696 }
697 return path;
698}

723//---------------------------------------------------------------------------------------------------------------------
724void VGrainlineItem::userMoveAndResize(const QPointF &pos)
725{
726 if (m_resizeHandle.containsPoint(pos, Qt::OddEvenFill) == true)
727 {
728 m_mode = Mode::Resize;
729 setCursor(Qt::SizeFDiagCursor);
730 }
731 else
732 {
